cmake_minimum_required (VERSION 2.6)

project(PCL_JAVA)

set(TARGET_NAME PCL_JAVA)
set(CMAKE_BINARY_DIR ${CMAKE_SOURCE_DIR})
set(LIBRARY_OUTPUT_DIR ${CMAKE_BINARY_DIR}/lib)

if (MSVC)
	set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_DEBUG ${LIBRARY_OUTPUT_DIR})
	set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_RELEASE ${LIBRARY_OUTPUT_DIR})
	set(CMAKE_RUNTIME_LIBRARY_DIRECTORY_DEBUG ${LIBRARY_OUTPUT_DIR})
	set(CMAKE_RUNTIME_LIBRARY_DIRECTORY_RELEASE ${LIBRARY_OUTPUT_DIR})
else()
	set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${LIBRARY_OUTPUT_DIR})
	set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${LIBRARY_OUTPUT_DIR})
endif()

find_package(JNI REQUIRED)

if (JNI_FOUND)
    message (STATUS "JNI_INCLUDE_DIRS=${JNI_INCLUDE_DIRS}")
	message (STATUS "JNI_LIBRARIES=${JNI_LIBRARIES}")
else()
	message(FATAL_ERROR "Java not found.")
endif()

find_package(PCL 1.8.1 REQUIRED)

if (PCL_FOUND)
	message(STATUS "PCL found.")
else()
	message(FATAL_ERROR "PCL not found.")
endif()

file(GLOB_RECURSE PCL_JAVA_SRC
	"${CMAKE_SOURCE_DIR}/src/main/c/*.h"
	"${CMAKE_SOURCE_DIR}/src/main/c/*.cpp"
) 

add_library(${TARGET_NAME} SHARED "${PCL_JAVA_SRC}")

target_include_directories(${TARGET_NAME} PUBLIC ${JNI_INCLUDE_DIRS} ${PCL_INCLUDE_DIRS})
target_link_libraries (${TARGET_NAME} ${PCL_COMMON_LIBRARIES} ${PCL_IO_LIBRARIES} ${JNI_LIBRARIES})
